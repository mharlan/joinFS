00099     _stacki_ --;\
00100     _start_ = _startStack_[_stacki_];\
00101     _end_ = _endStack_[_stacki_];\

00103       _p_ = _start_;\
00104       _i_ = _start_ + 1;\
00105       _j_ = _end_ - 1;\
00106       while (_i_<_j_) {\
00107         for(; _i_<=_j_ && comparator(((a)[_i_]),((a)[_p_]))<=0; _i_++) ;\
00108         if (_i_ > _j_) {\
00109           /* all remaining elements lesseq than pivot */\
00110           elem_exchanger(type, a, _j_, _p_);\
00111           _i_ = _j_;\
00112         } else {\
00113           for(; _i_<=_j_ && comparator(((a)[_j_]),((a)[_p_]))>=0; _j_--) ;\

00115             /* all remaining elements greater than pivot */\
00116             elem_exchanger(type, a, _j_, _p_);\

00143         } else {\

00246 /* ----------------- hashed table of pointers (in an array) -------------------- */
00247 
00248 /*
00249 
00250   This hashed table is storing pointers to objects (not containers).
00251   In this table there is a one-to-one mapping between 'objects' stored
00252   in the table and indexes where they are placed. Each index is
00253   pointing to exactly one 'object' and each 'object' stored in the
00254   table occurs on exactly one index.  Once an object is stored in the
00255   table, it can be represented via its index.
00256 
00257   In case of collision while adding an object the index shifted 
00258   by SGLIB_HASH_TAB_SHIFT_CONSTANT (constant can be redefined)
00259 
00260   You can NOT delete an element from such hash table. The only
00261   justification (I can see) for this data structure is an exchange
00262   file format, having an index table at the beginning and then
00263   refering objects via indexes.
00264 
00265   !!!!!!! This data structure is NOT documented, do not use it !!!!!!!!!! 
00266 
00267 */
00268

00270   int _i_;\
00271   for(_i_ = 0; _i_ < (dim); _i_++) (table)[_i_] = NULL;\
00272 }

00727 #define SGLIB___BIN_TREE_MAP_ON_ELEMENTS(type, tree, iteratedVariable, order, left, right, command) {\
00728   /* this is non-recursive implementation of tree traversal */\
00729   /* it maintains the path to the current node in the array '_path_' */\
00730   /* the _path_[0] contains the root of the tree; */\
00731   /* the _path_[_pathi_] contains the _current_element_ */\
00732   /* the macro does not use the _current_element_ after execution of command */\
00733   /* command can destroy it, it can free the element for example */\
00734   type *_path_[SGLIB_MAX_TREE_DEEP];\
00735   type *_right_[SGLIB_MAX_TREE_DEEP];\
00736   char _pass_[SGLIB_MAX_TREE_DEEP];\
00737   type *_cn_;\
00738   int _pathi_;\
00739   type *iteratedVariable;\
00740   _cn_ = (tree);\
00741   _pathi_ = 0;\
00742   while (_cn_!=NULL) {\
00743     /* push down to leftmost innermost element */\
00744     while(_cn_!=NULL) {\
00745       _path_[_pathi_] = _cn_;\
00746       _right_[_pathi_] = _cn_->right;\
00747       _pass_[_pathi_] = 0;\
00748       _cn_ = _cn_->left;\
00749       if (order == 0) {\
00750         iteratedVariable = _path_[_pathi_];\
00751         {command;}\
00752       }\
00753       _pathi_ ++;\
00754       if (_pathi_ >= SGLIB_MAX_TREE_DEEP) assert(0 && "the binary_tree is too deep");\
00755     }\
00756     do {\
00757       _pathi_ --;\
00758       if ((order==1 && _pass_[_pathi_] == 0)\
00759           || (order == 2 && (_pass_[_pathi_] == 1 || _right_[_pathi_]==NULL))) {\
00760         iteratedVariable = _path_[_pathi_];\
00761         {command;}\
00762       }\
00763       _pass_[_pathi_] ++;\
00764     } while (_pathi_>0 && _right_[_pathi_]==NULL) ;\
00765     _cn_ = _right_[_pathi_];\
00766     _right_[_pathi_] = NULL;\
00767     _pathi_ ++;\
00768   }\
00769 }

01415    }\
01416    if (ce != NULL) {\
01417      it->prevelem = ce->previous;\
01418    } else {\
01419      ce = it->nextelem;\
01420      it->nextelem = NULL;\
01421      if (it->subcomparator != NULL) {\
01422            eq = it->equalto; \
01423        scp = it->subcomparator;\
01424        while (ce!=NULL && scp(ce, eq)!=0) ce = ce->next;\
01425      }\
01426      if (ce != NULL) it->nextelem = ce->next;\
01427    }\
01428    it->currentelem = ce;\
01429    return(ce);\

01437 This  implementation requires  pointers  to left  and  right sons  (no
01438 parent  pointer  is needed)  and  one  bit  of additional  information
01439 storing the color of  the node. The implementation follows discrepancy
01440 fixing rules from:
01441 http://www.cis.ohio-state.edu/~gurari/course/cis680/cis680Ch11.html

01445 #define SGLIB___RBTREE_FIX_INSERTION_DISCREPANCY(type, tree, leftt, rightt, bits, RED, BLACK) {\
01446   type *t, *tl, *a, *b, *c, *ar, *bl, *br, *cl, *cr;\
01447   t = *tree;\
01448   tl = t->leftt;\
01449   if (t->rightt!=NULL && SGLIB___GET_VALUE(t->rightt->bits)==RED) {\
01450     if (SGLIB___GET_VALUE(tl->bits)==RED) {\
01451       if ((tl->leftt!=NULL && SGLIB___GET_VALUE(tl->leftt->bits)==RED) \
01452           || (tl->rightt!=NULL && SGLIB___GET_VALUE(tl->rightt->bits)==RED)) {\
01453         SGLIB___SET_VALUE(t->leftt->bits,BLACK);\
01454         SGLIB___SET_VALUE(t->rightt->bits,BLACK);\
01455         SGLIB___SET_VALUE(t->bits,RED);\
01456       }\
01457     }\
01458   } else {\
01459     if (SGLIB___GET_VALUE(tl->bits)==RED) {\
01460       if (tl->leftt!=NULL && SGLIB___GET_VALUE(tl->leftt->bits)==RED) {\
01461         a = t; b = tl; c = tl->leftt;\
01462         br = b->rightt;\
01463         a->leftt = br;\
01464         b->leftt = c; b->rightt = a;\
01465         SGLIB___SET_VALUE(a->bits,RED);\
01466         SGLIB___SET_VALUE(b->bits,BLACK);\
01467         *tree = b;\
01468       } else if (tl->rightt!=NULL && SGLIB___GET_VALUE(tl->rightt->bits)==RED) {\
01469         a = t; b = tl; ar=a->rightt;\
01470         bl=b->leftt; c=b->rightt;\
01471         cl=c->leftt; cr=c->rightt;\
01472         b->rightt = cl;\
01473         a->leftt = cr;\
01474         c->leftt = b;\
01475         c->rightt = a;\
01476         SGLIB___SET_VALUE(c->bits,BLACK);\
01477         SGLIB___SET_VALUE(a->bits,RED);\
01478         *tree = c;\
01479       }\
01480     }\
01481   }\

01484 #define SGLIB___RBTREE_FIX_DELETION_DISCREPANCY(type, tree, leftt, rightt, bits, RED, BLACK, res) {\
01485   type  *t, *a, *b, *c, *d, *ar, *bl, *br, *cl, *cr, *dl, *dr;\

01487   assert(t!=NULL);\
01488   ar = a->rightt;\
01489   b = t->leftt;\
01490   if (b==NULL) {\
01491     assert(SGLIB___GET_VALUE(t->bits)==RED);\
01492     SGLIB___SET_VALUE(t->bits,BLACK);\
01493     res = 0;\
01494   } else {\
01495     bl = b->leftt;\
01496     br = b->rightt;\
01497     if (SGLIB___GET_VALUE(b->bits)==RED) {\
01498       if (br==NULL) {\
01499         *tree = b;\

01501         b->rightt = a;\

01505         c = br;\

01510           *tree = b;\

01513           a->leftt = c;\
01514           SGLIB___SET_VALUE(c->bits,RED);\
01515           res = 0;\
01516         } else if (cl!=NULL && SGLIB___GET_VALUE(cl->bits)==RED) {\
01517           if (cr!=NULL && SGLIB___GET_VALUE(cr->bits)==RED) {\
01518             d = cr;\
01519             dl = d->leftt;\
01520             dr = d->rightt;\
01521             *tree = d;\
01522             SGLIB___SET_VALUE(d->bits,BLACK);\
01523             d->leftt = b;\
01524             c->rightt = dl;\
01525             d->rightt = a;\
01526             a->leftt = dr;\
01527             res = 0;\
01528           } else {\
01529             *tree = c;\
01530             c->leftt = b;\

01533             b->rightt = cl;\
01534             a->leftt = cr;\

01536             res = 0;\
01537           }\

01539           assert(cl==NULL || SGLIB___GET_VALUE(cl->bits)==BLACK);\
01540           d = cr;\
01541           dl = d->leftt;\
01542           dr = d->rightt;\
01543           *tree = d;\
01544           SGLIB___SET_VALUE(d->bits,BLACK);\
01545           d->leftt = b;\
01546           c->rightt = dl;\
01547           d->rightt = a;\
01548           a->leftt = dr;\
01549           res = 0;\
01550         } else {\
01551           assert(0);\
01552           res = 0;\
01553         }\
01554       }\
01555     } else {\
01556       if ((bl==NULL || SGLIB___GET_VALUE(bl->bits)==BLACK) && (br==NULL || SGLIB___GET_VALUE(br->bits)==BLACK)) {\
01557         res = (SGLIB___GET_VALUE(a->bits)==BLACK);\
01558         SGLIB___SET_VALUE(a->bits,BLACK);\
01559         SGLIB___SET_VALUE(b->bits,RED);\
01560       } else if (bl!=NULL && SGLIB___GET_VALUE(bl->bits)==RED) {\
01561         if (br==NULL || SGLIB___GET_VALUE(br->bits)==BLACK) {\
01562           *tree = b;\
01563           SGLIB___SET_VALUE(b->bits,SGLIB___GET_VALUE(a->bits));\
01564           SGLIB___SET_VALUE(a->bits,BLACK);\
01565           b->rightt = a;\
01566           a->leftt = br;\
01567           SGLIB___SET_VALUE(bl->bits,BLACK);\
01568           res = 0;\
01569         } else {\
01570           assert(bl!=NULL);\
01571           assert(br!=NULL);\
01572           assert(SGLIB___GET_VALUE(bl->bits)==RED);\
01573           assert(SGLIB___GET_VALUE(br->bits)==RED);\
01574           c = br;\
01575           cl = c->leftt;\
01576           cr = c->rightt;\
01577           *tree = c;\
01578           SGLIB___SET_VALUE(c->bits,SGLIB___GET_VALUE(a->bits));\
01579           SGLIB___SET_VALUE(a->bits,BLACK);\
01580           c->leftt = b;\
01581           c->rightt = a;\
01582           b->rightt = cl;\
01583           a->leftt = cr;\
01584           res = 0;\
01585         }\
01586       } else {\
01587         assert(br!=NULL && SGLIB___GET_VALUE(br->bits)==RED);\
01588         c = br;\
01589         cl = c->leftt;\
01590         cr = c->rightt;\
01591         *tree = c;\
01592         SGLIB___SET_VALUE(c->bits,SGLIB___GET_VALUE(a->bits));\

01594         c->leftt = b;\
01595         c->rightt = a;\

01599       }\
01600     }\
01601   }\
