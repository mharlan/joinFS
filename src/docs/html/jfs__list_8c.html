Destroy a jfs list

Parameters:
head The list head.
op The type result in the jfs list. 

{
  struct sglib_jfs_list_t_iterator it;
  jfs_list_t *item;
  
  for(item = sglib_jfs_list_t_it_init(&it, head);
          item != NULL; item = sglib_jfs_list_t_it_next(&it)) {
        switch(op) {
        case(jfs_listattr_op):
          free(item->key);
          break;
        case(jfs_readdir_op):
          if(item->datapath) {
                free(item->datapath);
          }
      if(item->filename) {
        free(item->filename);
      }
        default:
          break;
        }

    free(item);
  }
}
