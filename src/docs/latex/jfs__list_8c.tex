  for(item = sglib_jfs_list_t_it_init(&it, head);
          item != NULL; item = sglib_jfs_list_t_it_next(&it)) {
        switch(op) {
        case(jfs_listattr_op):
          free(item->key);
          break;
        case(jfs_readdir_op):

      }
        default:
          break;
        }
